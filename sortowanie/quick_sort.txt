import random

def quick_sort(arr, left, right, pivot_type='left'):
    if left < right:
        pivot_index = partition(arr, left, right, pivot_type)
        quick_sort(arr, left, pivot_index - 1, pivot_type)
        quick_sort(arr, pivot_index + 1, right, pivot_type)

def partition(arr, left, right, pivot_type):
    if pivot_type == 'random':
        pivot_index = random.randint(left, right)
        arr[left], arr[pivot_index] = arr[pivot_index], arr[left]
    
    pivot = arr[left]
    i = left + 1
    
    for j in range(left + 1, right + 1):
        if arr[j] < pivot:
            arr[i], arr[j] = arr[j], arr[i]
            i += 1
    
    arr[left], arr[i - 1] = arr[i - 1], arr[left]
    return i - 1

quick_sort(arr, 0, len(arr) - 1, pivot_type='random')
